# Nginx

### Activate website
ln -s /etc/nginx/sites-available/memes.steffbeckers.eu /etc/nginx/sites-enabled/memes.steffbeckers.eu

### Reload nginx
sudo systemctl reload nginx


# Certbot
### Installation
sudo add-apt-repository ppa:certbot/certbot
sudo apt-get update
sudo apt-get install python-certbot-nginx

### Wildcard certificate
certbot -d steffbeckers.eu -d *.steffbeckers.eu --manual --preferred-challenges dns --server https://acme-v02.api.letsencrypt.org/directory certonly
